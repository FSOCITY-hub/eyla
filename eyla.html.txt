function nextImage(id){

// =================== CAROUSEL ===================
let slideIndex={};
function nextSlide(id){
  const carousel=document.getElementById("carousel-"+id).querySelector(".carousel-inner");
  const slides=carousel.children.length;
  slideIndex[id]=slideIndex[id]===undefined?0:slideIndex[id];
  slideIndex[id]=(slideIndex[id]+1)%slides;
  carousel.style.transform=`translateX(-${slideIndex[id]*100}%)`;
}
function prevSlide(id){
  const carousel=document.getElementById("carousel-"+id).querySelector(".carousel-inner");
  const slides=carousel.children.length;
  slideIndex[id]=slideIndex[id]===undefined?0:slideIndex[id];
  slideIndex[id]=(slideIndex[id]-1+slides)%slides;
  carousel.style.transform=`translateX(-${slideIndex[id]*100}%)`;
}
